355 fun bagPacking(treasures: Map<String, Pair<Int, Int>>, capacity: Int): Set<String> {
356     //Used the information here: https://neerc.ifmo.ru/wiki/index.php?title=%D0%97%D0%B0%D0%B4%D0%B0%D1%87%D0%B0_%D0%BE_%D1%80%D1%8E%D0%BA%D0%B7%D0%B0%D0%BA%D0%B5
357     val names = treasures.keys.toMutableList()
358     val weightAndPrice = treasures.values.toMutableList()
359     val numberOfItems = treasures.size
360     val bag = Array(numberOfItems + 1) { IntArray(capacity + 1) }
361     weightAndPrice.forEachIndexed { index, (weight, price) ->
362         for (s in 1..capacity) {
363             if (s >= weight) {
364                 bag[index + 1][s] = max(bag[index][s], bag[index][s - weight] + price)
365             } else bag[index + 1][s] = bag[index][s]
366         }
367     }
368     val result = mutableSetOf<String>()
369     var s = capacity
370     for (k in numberOfItems downTo 1) {
371         if (bag[k - 1][s] != bag[k][s]) {
372             result.add(names[k - 1])
373             s -= weightAndPrice[k - 1].first
374         }
375     }
376     return result
377 }
